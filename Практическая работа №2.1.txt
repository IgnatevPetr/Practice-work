#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

int main() {
  double a, b, n, m;
  cout << "Введите числа a и b: ";
  cin >> a >> b;
  cin.ignore();
  if (a >= b)
    n = pow((a-b), (1.0/3));
  else
    n = a*a + (a-b)/sin(a*b);
  if (n < b)
    m = (n+a)/-1*b+sqrt(pow(sin(a), 2)-cos(n));
  else  if (n == b)
          m = b*b + tan(n*a);
        else
          m = b*b*b + n*a*a;
cout << "Числа после преобразований:\n";
cout << "a = " << a << endl;
cout << "b = " << b << endl;
cout << fixed;
cout << "n = " << setprecision(3) << n << endl;
cout << "m = " << setprecision(3) << m << endl;
cin.get();
}
