#include <iostream>
#include <cstdlib>
#include <math.h>

using namespace std;

class vector{
  private:
    double x, y;
  public:
    void put_x(){
      cin >> x;
    }
    void put_y(){
      cin >> y;
    }
    void get_info(){
      cout << "(" << x << ", " << y << ")" << endl;
    }
};

int main (int argc, char *argv[]){
  vector a, b;
  cout << "Координата x вектора a: ";
  a.put_x();
  cout << "Координата y вектора a: ";
  a.put_y();
  cout << "Координата x вектора b: ";
  b.put_x();
  cout << "Координата y вектора b: ";
  b.put_y();
  cout << "\n" << "Векторы:" << endl;
  cout << "a = ";
  a.get_info();
  cout << "b = ";
  b.get_info();
  return EXIT_SUCCESS;
}
