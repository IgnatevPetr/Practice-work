#include <iostream>
#include <iostream>
#include <cstdlib>
#include <string>

using namespace std;

class library {
public:
	struct book {
		string title;
		int COP; //Count of pages
	};
	void put_COB();
	void put_books();
	void get_info();
private:
	int COB; //Count of books
	book books[10];
};

void library::put_COB() {
	cout << "Введите количество книг: ";
	cin >> COB;
}

void library::put_books() {
	for (int i = 0; i < COB; i++) {
		cout << "Введите название книги " << i + 1 << ": ";
		cin.ignore();
		getline(cin, books[i].title);
		cout << "Введите кол-во страниц книги " << i + 1 << ": ";
		cin.ignore();
		cin >> books[i].COP;
	}
}

void library::get_info() {
  cout << "\nБиблиотека:" << endl;
	for (int k = 0; k < COB; k++) {
		cout << "Книга " << k + 1 << ": " << books[k].title << " (" << books[k].COP << " стр.)" << endl;
	}
}

int main(int argc, char *argv[]) {
	library lib1;
	lib1.put_COB();
	lib1.put_books();
	lib1.get_info();
	return EXIT_SUCCESS;
}
