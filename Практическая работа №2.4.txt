#include <iostream>
#include <cmath>
#include <iomanip>
#define _USE_MATH_DEFINES
using namespace std;

int main() {
  double a, x, y, t;
  cout << "Введите числа a и b: ";
  cin >> a >> x;
  cin.ignore();
  if (a <= x)
    y = a+log(x+a);
  else
    y = sqrt(sin(a*x*M_PI/180));
  if (a > y)
    t = y/(a-x);
  else  if (a == y)
          t = y/(a-x)+(a+x)/y/y;
        else
          t = tan(a*x*M_PI/180)+cos(2*a*y*M_PI/180);
cout << "Числа после преобразований:\n";
cout << "a = " << a << endl;
cout << "x = " << x << endl;
cout << fixed;
cout << "y = " << setprecision(3) << y << endl;
cout << "t = " << setprecision(3) << t << endl;
cin.get();
}
