#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

int main() {
  double a, b, z, t;
  cout << "Введите числа a и b: ";
  cin >> a >> b;
  cin.ignore();
  if (a < b)
    z = sqrt(abs(a*a-b*b));
  else
    z = 1-2*cos(a)*sin(b);
  if (z < b)
    t = pow(z+a*a*b, 1.0/3);
  else  if (z == b)
          t = 1-log(z)+cos(a*a*b);
        else
          t = 1/cos(z*a);
cout << "Числа после преобразований:\n";
cout << "a = " << a << endl;
cout << "b = " << b << endl;
cout << fixed;
cout << "z = " << setprecision(3) << z << endl;
cout << "t = " << setprecision(3) << t << endl;
cin.get();
}
